\newcommand{\coverpage}{
	\thispagestyle{empty}
	\pagenumbering{gobble}
}

\newcommand{\toc}{
	\newpage
	\pagenumbering{Roman}
	\setcounter{page}{1}
	\tableofcontents
}

\newcommand{\maincontents}{
	\newpage
	\pagestyle{fancy}
	\setlength{\headheight}{12.2pt}
	\setlength{\headsep}{15pt}
	\pagenumbering{arabic}
	\setcounter{page}{1}
}
