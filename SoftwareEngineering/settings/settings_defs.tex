% the code environment
\newenvironment{code}[2]	% lanuage, gobble-char-num
{	\codeF
	\lstset{
		language=#1
		tabsize=4,
		breaklines=true,
		frame=shadowbox,
		framexleftmargin=7mm,
		rulesepcolor=\color{black},
		basicstyle=\small,
		numbers=left
		gobble=#2
	}
	\begin{lstlistings}
}

{\end{lstlistings}}
